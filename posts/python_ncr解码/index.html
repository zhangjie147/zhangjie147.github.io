NCR解码&#xFFFF

02-08-2022

  • Python
  • NCR

NCR编码有&#开头和&#x开头

先将&#开头十进制全部转换为&#x开头十六进制的

如&#16 —》&#x10
    # 找到&#开头
    re.sub("&#(?P<number>\d+)",NCR._strChange10to16,line) 
    # 转换进制
    def _strChange10to16(matched):
        intStr = matched.group("number")
        intStr = '&#x' + hex(int(intStr,10)).replace(r'0x','').upper()
        return intStr

将不是四位&#x开头十六进制编码补全为四位的

如&#xFF,&#xFFF —》&#x00FF,&#x0FFF
    line = re.sub(r'&#x([A-F0-9]{2});', r'&#x00\1;', line)    
    line = re.sub(r'&#x([A-F0-9]{3});', r'&#x0\1;', line)

使用Python的encode(),decode()函数转化

    _t = _t.replace(';','')\ # 去掉";"
           .replace('&#x','\\u')\ # '&#x' 替换为 '\\u'
           .encode('utf-8')\ # 转码
           .decode('unicode_escape')

完整代码

#!/usr/bin/python
import re,os
class NCR():
    def _strChange10to16(matched):
        intStr = matched.group("number")
        intStr = '&#x' + hex(int(intStr,10)).replace(r'0x','').upper()
        return intStr

    # 整段转换编码 我不会区分NCR编码外的其他能容 用下面的
    # def dec(_t):
    #     _t1 = re.sub("&#(?P<number>\d+)",NCR._strChange10to16,_t)
    #     _t1 = re.sub(r'&#x([A-F0-9]{2});', r'&#x00\1;', _t1)
    #     _tt = _t1.replace(';','')\
    #             .replace('&#x','\\u')\
    #             .encode('utf-8')\
    #             .decode('unicode_escape')
    #     #print(_tt)
    #     return _tt

    # 单个字转换编码        
    def dec_one(matched):
        intStr = matched.group("number")
        intStr = intStr.replace(';','')\
                       .replace('&#x','\\u')\
                       .encode('utf-8')\
                       .decode('unicode_escape')
        #print(intStr)
        return intStr

    def updateFile(file):
        
        with open(file, "r", encoding="utf-8") as f1,open("%s.bak" % file, "w", encoding="utf-8") as f2:
            for line in f1:
                # 将 &# 开头十进制转换为 &#x 开头十六进制
                line = re.sub("&#(?P<number>\d+)",NCR._strChange10to16,line) 
                # 补全 &#x FF 到 &#x 00FF 四位十六进制 
                line = re.sub(r'&#x([A-F0-9]{2});', r'&#x00\1;', line)    
                # 补全 &#x FFF 到 &#x 0FFF 四位十六进制  我是新手就会这样写了      
                line = re.sub(r'&#x([A-F0-9]{3});', r'&#x0\1;', line)  
                # 查找所有 &#x 开头 NCR 编码 转换为 文字 utf-8      
                line = re.sub(r'(?P<number>&#x([A-Fa-f0-9]{4});)', NCR.dec_one, line) 
                f2.write(line)
        os.remove(file)
        os.rename("%s.bak" % file, file)
    
NCR.updateFile(r"文件路径")
